// Script JavaScript pour afficher les options spécifiques aux professeurs
document.getElementById('formStatut').addEventListener('submit', function(event) {
	event.preventDefault(); // Empêcher le formulaire de se soumettre

	var statut = document.getElementById('statut').value;

	// Masquer toutes les sections spécifiques
	document.querySelectorAll('[id^="options"]').forEach(function(element) {
    	element.style.display = 'none';
	});

	// Afficher la section spécifique au statut sélectionné
	if (statut === 'professeur') {
    	document.getElementById('optionsProfesseur').style.display = 'block';
	} else if (statut === 'etudiant') {
    	document.getElementById('optionsEtudiant').style.display = 'block';
	}
});

	event.preventDefault(); // Empêcher le formulaire de se soumettre

        	var ufr = document.getElementById('ufr').value;

        	// Masquer tous les formulaires de matières
        	document.querySelectorAll('[id^="matieres"]').forEach(function(element) {
            	element.style.display = 'none';

        	document.getElementById('matieres' + ufr).style.display = 'block';
    	});

    	document.getElementById('nombreMatieres').addEventListener('change', function(event) {
        	var nombreMatieres = parseInt(this.value);
        	var checkboxes = document.querySelectorAll('[id^="matieres"] input[type="checkbox"]');

        	// Réinitialiser toutes les cases à cocher
        	checkboxes.forEach(function(checkbox) {
            	checkbox.checked = false;
        	});

        	// Activer les cases à cocher correspondant au nombre choisi de matières
        	for (var i = 0; i < nombreMatieres; i++) {
            	checkboxes[i].disabled = false;
        	}

        	// Afficher le formulaire de choix des matières
        	document.getElementById('formMatieres').style.display = 'block';
    	});
